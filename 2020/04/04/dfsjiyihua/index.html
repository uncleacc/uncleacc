using namespace std;
const int maxn=110;
int map[maxn][maxn];
int step[maxn][maxn];
int Next[][2]={{1,0},{0,1},{-1,0},{0,-1}};
int m,n;
int judge(int x,int y){
	if(x>=0&&x<m&&y>=0&&y<n) return 1;
	else return 0;
}
int dfs(int x,int y){
	if(step[x][y]) return step[x][y];
	for(int i=0;i<4;i++){
		int tx=x+Next[i][0];
		int ty=y+Next[i][1];
		if(judge(tx,ty)&&map[tx][ty]<map[x][y]){
			int temp=dfs(tx,ty)+1;
			step[x][y]=max(temp,step[x][y]);
		}
	}
	return step[x][y];
}

	int ans=0;
	cin>>m>>n;

		}

		for(int j=0;j<n;j++){
			step[i][j]=dfs(i,j);
			if(ans<step[i][j]) ans=step[i][j];
		}
	}

}
