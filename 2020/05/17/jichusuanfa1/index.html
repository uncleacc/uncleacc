using namespace std;
const int MAXN=1e5+100;
int val[MAXN];
int main()
{
	ios;
	int t,kase=0; cin>>t;
	while(t--){
		memset(val,0,sizeof val);
		int n; cin>>n;
		for(int i=1;i<=n;i++) cin>>val[i];
		int summax=-1e9,b=0,e=0,k=1,sum=0; 
		for(int i=1;i<=n;i++){
			sum+=val[i];
			if(sum>summax){
				b=k;
				e=i;
				summax=sum;
			}
			if(sum<0){
				sum=0;
				k=i+1;
			}
		}
		printf("Case %d:\n",++kase);
		cout<<summax<<" "<<b<<" "<<e<<endl;
//		printf("Case %d:\n%d %d %d\n",++kase,summax,b,e);
		if(t!=0) printf("\n");
//		if(t!=0) cout<<endl; 
	} 
	return 0;
}
