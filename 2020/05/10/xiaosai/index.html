	if(x+1>n||x+1<1||y+1>m||y+1<1) return 0;

	else{
		s[x][y]='#';

{

	while(scanf("%d %d",&m,&n)!=EOF){

char map[9][9];
int vis[50],n,k,ans;
void dfs(int m,int x)
{
    if(x==k)
    {
//    	printf("%d\n",m);
        ans++;
        return ;
    }
    for(int i=m;i<=n;i++)
        for(int j=0;j<n;j++)
    {
        if(map[i][j]=='.'||vis[j]==1) continue;
        vis[j]=1;
        dfs(i+1,x+1);
        vis[j]=0;
    }
}
int main()
{
    while(~scanf("%d%d",&n,&k))
    {
        if(n==-1&&k==-1)
            break;
        for(int i=1;i<=n;i++)
            scanf("%s",&map[i]);
        memset(vis,0,sizeof(vis));
        ans=0;
        dfs(1,0);
        printf("%d\n",ans);
    }
    return 0;
}
