{
	phi[1]=1;
	for(int i=2;i<MAXN;i++){
		if(!isp[i]){
			pr[++tail]=i;
			phi[i]=i-1;

			while(n%i==0) n/=i;
		}
	}
	if(n>1) ans=ans/n*(n-1);
	return ans;
}
